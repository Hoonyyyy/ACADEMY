### view 이용
```mysql
# 학과평균나이 뷰를 생성하시오.
# 학생테이블 소속학과를 기준으로 학과의 평균나이를 출력해준다.

CREATE VIEW 학과평균나이 AS
SELECT truncate(AVG(나이), 0) AS 평균나이
FROM 학생
GROUP BY 소속학과;
```

### INSET INTO는 하기 힘드나, UPDATE SET은 쉽게 가능함
```mysql
# 컴퓨터과학과 C303교수의 주소를 경기도 의정부로 수정하시오.
UPDATE 컴퓨터과학과교수
set 주소 = '경기도 의정부'
where 교수번호 = 'C303';
```
