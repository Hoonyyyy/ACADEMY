```java
		while(true) {
			
			int num = sc.nextInt();
```

## do - while은.. 보고만 넘겼는데 하긴 해야겠다
```java
package ch05_반복문;

import java.util.Scanner;

// 숫자 맞추기 게임

public class DoWhile3 {
	
	public static void main(String[] args) {
		
		System.out.println("1~100중에서 어떤 숫자일까요?");
		
		Scanner sc = new Scanner(System.in);
		
		int answer = (int)(Math.random()*100)+1;
//		System.out.println(answer);
		int cnt = 0;
		int num;
		
		do {
			cnt++;
			
			num = sc.nextInt();
			if(answer == num) {
				System.out.println("축하합니다." + cnt + "번 만에 맞췄습니다.");
			}
			else if(answer > num) {
				System.out.println("더 큰 수를 입력하세요");
			}
			else if(answer < num) {
				System.out.println("더 작은 수를 입력하세요");
			}
			
		}while(cnt<10);
		
		System.out.println("10번 넘겼어요");
		
	}
}
```

```java
	Account(int amount) {
		this.amount = amount;
	}
	
	// 입금
	public int deposit(int money) {
		
		this.amount += money;
		System.out.println("현재 잔액은" + amount + "원 입니다.");
		return amount;
	}
	
	// 출금
	public int withdraw(int money) {
		if (this.amount < money) {
			System.out.println("잔액이 부족합니다.");
		} else {
			this.amount -= money;
			System.out.println(money +"원을 출금 하셨습니다.");
			System.out.println("현재 잔액은 " + amount + "원 입니다.");
		}
		return amount;
	}
	
	
	// 잔고 확인
	public int check(int money) {
		System.out.println("현재 잔액은" + amount + "원 입니다.");
		return amount;
	}
	
	
	
	
	public static void main(String[] args) {
		
		Account B = new Account(0);  // amount 값 0
		int money = 0;
		
		Scanner sc = new Scanner(System.in);
		boolean exit = false;
		
		do {
			System.out.println(
					"----------------------\n"
```
