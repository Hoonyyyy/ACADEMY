### Rect , strokeRect
```html
<!DOCTYPE html>
<!-- 
	rect(x,y,w,h)	-	사각형 경로에 x,y,w,h 적용
	strokeRect(x,y,w,h)	-	캔버스에 직접그린다.	
 -->
<html>
<head>
<meta charset="UTF-8">
<title>Insert title here</title>
</head>
<body>
	<canvas
		id="myCanvas"
		style="background: aliceblue;"
		width="500"
		height="500"
	></canvas>
	<script>
		let canvas = document.getElementById('myCanvas');
		let context = canvas.getContext('2d');


		// 경로를 만들어서 그린 사각형
		context.beginPath();
		context.strokeStyle="red";
		for(i=0;i<10;i++){
			context.rect(10+i*40,10+i*10,50,50);
		}

		context.stroke();

		// 외곽선이 있는 사각형
		context.lineWidth = 3;
		context.strokeStyle = "green";
		context.strokeRect(50,250,100,100);

	</script>
</body>
</html>
```

### 칠판 쓰듯이.. 근데 코드 이해가 잘 안됨
```html
<!DOCTYPE html>
<html>
<head>
<meta charset="UTF-8">
<title>Insert title here</title>
</head>
<body onload="init()">
	<canvas
		id="myCanvas"
		width="1800"
		height="800"
		style="background: green; border: 20px solid #cd853f;
				text-align: center;"
	></canvas>
	<script>
		let canvas, context;
		function init(){
			canvas = document.getElementById('myCanvas');
			canvas.style.width = window.innerWidth;
			context = canvas.getContext('2d');
			
			context.lineWidth = 2;
			context.strokeStyle = 'white';

			canvas.addEventListener('mousemove',move,false);
			canvas.addEventListener('mousedown',down,false);
			canvas.addEventListener('mouseup',up,false);
			canvas.addEventListener('mouseout',out,false);

		}

		let startX=0;
		let startY=0;
		let dragging = false;

		function draw(curX, curY){
			context.beginPath();
			context.moveTo(startX,startY);
			context.lineTo(curX,curY);
			context.stroke();
		}

		function move(e){

			if(!dragging){
				return;
			}
			let curX=e.offsetX, curY=e.offsetY;
			draw(curX,curY);

			startX = curX;
			startY = curY;
		}

		function down(e){
			startX = e.offsetX;
			startY = e.offsetY;
			dragging = true;
		}

		function up(){
			dragging = false;
		}

		function out(){
			dragging = false;
		}


	</script>
</body>
</html>

```
