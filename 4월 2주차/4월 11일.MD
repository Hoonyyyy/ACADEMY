	}

		font-size: 20px;

	}

		cursor: pointer;

		font-weight: 900;
	}

		text-align: center;
	}

	function greet(){
		document.write("<h1>환영합니다<h1>");
			
	}

<body>
	<div>
		<p><script>greet()</script></p>
		<hr>
